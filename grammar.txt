# This is a raw grammar for the parser generator grammar grammar.

grammar
    : ( rule )+

rule
    : ':' production ( '|' production )* ';'

production
    : ( prod_elem )+

prod_elem
    : TERMINAL
    | NON_TERMINAL
    | zero_or_more
    | one_or_more
    | one_or_zero
    | group

zero_or_more
    : group '*'

one_or_more
    : group '+'

one_or_zero
    : group '?'

group
    : '(' (prod_elem)+ ')'
